57 y = iterate(itr)
58 y === nothing && return true

60 y = iterate(itr, state)

66 end
67 return true

373 """
374 PartialQuickSort{T <: Union{Int,OrdinalRange}}
375
376 Indicate that a sorting function should use the partial quick sort
377 algorithm. Partial quick sort returns the smallest `k` elements sorted from smallest
378 to largest, finding them and sorting them using [`QuickSort`](@ref).
379
380 Characteristics:
381 * *not stable*: does not preserve the ordering of elements which
382 compare equal (e.g. "a" and "A" in a sort of letters which
383 ignores case).
384 * *in-place* in memory.
385 * *divide-and-conquer*: sort strategy similar to [`MergeSort`](@ref).
386 """
387 struct PartialQuickSort{T <: Union{Int,OrdinalRange}} <: Algorithm
388 k::T
389 end
390
391 Base.first(a::PartialQuickSort{Int}) = 1
392 Base.last(a::PartialQuickSort{Int}) = a.k
393 Base.first(a::PartialQuickSort) = first(a.k)
394 Base.last(a::PartialQuickSort) = last(a.k)

396 """
397 InsertionSort
398
399 Indicate that a sorting function should use the insertion sort
400 algorithm. Insertion sort traverses the collection one element
401 at a time, inserting each element into its correct, sorted position in
402 the output list.
403
404 Characteristics:
405 * *stable*: preserves the ordering of elements which
406 compare equal (e.g. "a" and "A" in a sort of letters
407 which ignores case).
408 * *in-place* in memory.
409 * *quadratic performance* in the number of elements to be sorted:
410 it is well-suited to small collections but should not be used for large ones.
411 """

414 QuickSort
415
416 Indicate that a sorting function should use the quick sort

418
419 Characteristics:
420 * *not stable*: does not preserve the ordering of elements which
421 compare equal (e.g. "a" and "A" in a sort of letters which

468 # selectpivot!
469 #
470 # Given 3 locations in an array (lo, mi, and hi), sort v[lo], v[mi], v[hi]) and
471 # choose the middle value as a pivot
472 #
473 # Upon return, the pivot is in v[lo], and v[hi] is guaranteed to be
474 # greater than the pivot
475
476 @inline function selectpivot!(v::AbstractVector, lo::Int, hi::Int, o::Ordering)
477 @inbounds begin
478 mi = (lo+hi)>>>1
479
480 # sort the values in v[lo], v[mi], v[hi]
481
482 if lt(o, v[mi], v[lo])
483 v[mi], v[lo] = v[lo], v[mi]
484 end
485 if lt(o, v[hi], v[mi])
486 if lt(o, v[hi], v[lo])
487 v[lo], v[mi], v[hi] = v[hi], v[lo], v[mi]
488 else
489 v[hi], v[mi] = v[mi], v[hi]
490 end
491 end
492
493 # move v[mi] to v[lo] and use it as the pivot
494 v[lo], v[mi] = v[mi], v[lo]
495 pivot = v[lo]
496 end
497
498 # return the pivot
499 return pivot
500 end

502 # partition!
503 #
504 # select a pivot, and partition v according to the pivot
505
506 function partition!(v::AbstractVector, lo::Int, hi::Int, o::Ordering)
507 pivot = selectpivot!(v, lo, hi, o)
508 # pivot == v[lo], v[hi] > pivot
509 i, j = lo, hi
510 @inbounds while true
511 i += 1; j -= 1
512 while lt(o, v[i], pivot); i += 1; end;
513 while lt(o, pivot, v[j]); j -= 1; end;
514 i >= j && break
515 v[i], v[j] = v[j], v[i]
516 end
517 v[j], v[lo] = pivot, v[j]
518
519 # v[j] == pivot
520 # v[k] >= pivot for k > j
521 # v[i] <= pivot for i < j
522 return j
523 end

642 """
643 sort!(v; alg::Algorithm=defalg(v), lt=isless, by=identity, rev::Bool=false, order::Ordering=Forward)
644
645 Sort the vector `v` in place. `QuickSort` is used by default for numeric arrays while
646 `MergeSort` is used for other arrays. You can specify an algorithm to use via the `alg`
647 keyword (see Sorting Algorithms for available algorithms). The `by` keyword lets you provide
648 a function that will be applied to each element before comparison; the `lt` keyword allows
649 providing a custom "less than" function; use `rev=true` to reverse the sorting order. These
650 options are independent and can be used together in all possible combinations: if both `by`
651 and `lt` are specified, the `lt` function is applied to the result of the `by` function;
652 `rev=true` reverses whatever ordering specified via the `by` and `lt` keywords.
653
654 # Examples
655 ```jldoctest
656 julia> v = [3, 1, 2]; sort!(v); v
657 3-element Array{Int64,1}:
658 1
659 2
660 3
661
662 julia> v = [3, 1, 2]; sort!(v, rev = true); v
663 3-element Array{Int64,1}:
664 3
665 2
666 1
667
668 julia> v = [(1, "c"), (3, "a"), (2, "b")]; sort!(v, by = x -> x[1]); v
669 3-element Array{Tuple{Int64,String},1}:
670 (1, "c")
671 (2, "b")
672 (3, "a")
673
674 julia> v = [(1, "c"), (3, "a"), (2, "b")]; sort!(v, by = x -> x[2]); v
675 3-element Array{Tuple{Int64,String},1}:
676 (3, "a")
677 (2, "b")
678 (1, "c")
679 ```
680 """
681 function sort!(v::AbstractVector;
682 alg::Algorithm=defalg(v),
683 lt=isless,
684 by=identity,
685 rev::Union{Bool,Nothing}=nothing,
686 order::Ordering=Forward)
687 ordr = ord(lt,by,rev,order)
688 if ordr === Forward && isa(v,Vector) && eltype(v)<:Integer
689 n = length(v)
690 if n > 1
691 min, max = extrema(v)
692 (diff, o1) = sub_with_overflow(max, min)
693 (rangelen, o2) = add_with_overflow(diff, oneunit(diff))
694 if !o1 && !o2 && rangelen < div(n,2)
695 return sort_int_range!(v, rangelen, min)

702 # sort! for vectors of few unique integers
703 function sort_int_range!(x::Vector{<:Integer}, rangelen, minval)
704 offs = 1 - minval
705 n = length(x)
706
707 where = fill(0, rangelen)
708 @inbounds for i = 1:n
709 where[x[i] + offs] += 1
710 end
711
712 idx = 1
713 @inbounds for i = 1:rangelen
714 lastidx = idx + where[i] - 1
715 val = i-offs
716 for j = idx:lastidx
717 x[j] = val
718 end
719 idx = lastidx + 1
720 end
721
722 return x
723 end

779 2
780 ```

810 ## sortperm: the permutation to sort an array ##
811
812 """
813 sortperm(v; alg::Algorithm=DEFAULT_UNSTABLE, lt=isless, by=identity, rev::Bool=false, order::Ordering=Forward)
814
815 Return a permutation vector `I` that puts `v[I]` in sorted order. The order is specified
816 using the same keywords as `sort!`. The permutation is guaranteed to be stable even if the
817 sorting algorithm is unstable, meaning that indices of equal elements appear in ascending
818 order.
819
820 See also [`sortperm!`](@ref).
821
822 # Examples
823 ```jldoctest
824 julia> v = [3, 1, 2];
825
826 julia> p = sortperm(v)
827 3-element Array{Int64,1}:
828 2
829 3
830 1
831
832 julia> v[p]
833 3-element Array{Int64,1}:
834 1
835 2
836 3
837 ```
838 """
839 function sortperm(v::AbstractVector;
840 alg::Algorithm=DEFAULT_UNSTABLE,
841 lt=isless,
842 by=identity,
843 rev::Union{Bool,Nothing}=nothing,
844 order::Ordering=Forward)
845 ordr = ord(lt,by,rev,order)
846 if ordr === Forward && isa(v,Vector) && eltype(v)<:Integer
847 n = length(v)
848 if n > 1
849 min, max = extrema(v)
850 (diff, o1) = sub_with_overflow(max, min)
851 (rangelen, o2) = add_with_overflow(diff, oneunit(diff))
852 if !o1 && !o2 && rangelen < div(n,2)
853 return sortperm_int_range(v, rangelen, min)
854 end
855 end
856 end
857 p = similar(Vector{Int}, axes(v, 1))
858 for (i,ind) in zip(eachindex(p), axes(v, 1))
859 p[i] = ind
860 end
861 sort!(p, alg, Perm(ordr,v))
862 end

866 sortperm!(ix, v; alg::Algorithm=DEFAULT_UNSTABLE, lt=isless, by=identity, rev::Bool=false, order::Ordering=Forward, initialized::Bool=false)
867
868 Like [`sortperm`](@ref), but accepts a preallocated index vector `ix`. If `initialized` is `false`
869 (the default), `ix` is initialized to contain the values `1:length(v)`.
870
871 # Examples
872 ```jldoctest
873 julia> v = [3, 1, 2]; p = zeros(Int, 3);
874
875 julia> sortperm!(p, v); p
876 3-element Array{Int64,1}:
877 2
878 3
879 1
880
881 julia> v[p]
882 3-element Array{Int64,1}:
883 1
884 2
885 3
886 ```

888 function sortperm!(x::AbstractVector{<:Integer}, v::AbstractVector;
889 alg::Algorithm=DEFAULT_UNSTABLE,
890 lt=isless,
891 by=identity,
892 rev::Union{Bool,Nothing}=nothing,
893 order::Ordering=Forward,
894 initialized::Bool=false)
895 if axes(x,1) != axes(v,1)
896 throw(ArgumentError("index vector must have the same indices as the source vector, $(axes(x,1)) != $(axes(v,1))"))
897 end
898 if !initialized
899 @inbounds for i = axes(v,1)
900 x[i] = i
901 end
902 end
903 sort!(x, alg, Perm(ord(lt,by,rev,order),v))
904 end

906 # sortperm for vectors of few unique integers
907 function sortperm_int_range(x::Vector{<:Integer}, rangelen, minval)
908 offs = 1 - minval
909 n = length(x)
910
911 where = fill(0, rangelen+1)
912 where[1] = 1
913 @inbounds for i = 1:n
914 where[x[i] + offs + 1] += 1
915 end
916
917 #cumsum!(where, where)
918 @inbounds for i = 2:length(where)
919 where[i] += where[i-1]
920 end
921
922 P = Vector{Int}(undef, n)
923 @inbounds for i = 1:n
924 label = x[i] + offs
925 P[where[label]] = i
926 where[label] += 1
927 end
928
929 return P
930 end

932 ## sorting multi-dimensional arrays ##
933
934 """
935 sort(A; dims::Integer, alg::Algorithm=DEFAULT_UNSTABLE, lt=isless, by=identity, rev::Bool=false, order::Ordering=Forward)
936
937 Sort a multidimensional array `A` along the given dimension.
938 See [`sort!`](@ref) for a description of possible
939 keyword arguments.
940
941 # Examples
942 ```jldoctest
943 julia> A = [4 3; 1 2]
944 2×2 Array{Int64,2}:
945 4 3
946 1 2
947
948 julia> sort(A, dims = 1)
949 2×2 Array{Int64,2}:
950 1 2
951 4 3
952
953 julia> sort(A, dims = 2)
954 2×2 Array{Int64,2}:
955 3 4
956 1 2
957 ```
958 """
959 function sort(A::AbstractArray;
960 dims::Integer,
961 alg::Algorithm=DEFAULT_UNSTABLE,
962 lt=isless,
963 by=identity,
964 rev::Union{Bool,Nothing}=nothing,
965 order::Ordering=Forward)
966 dim = dims
967 order = ord(lt,by,rev,order)
968 n = length(axes(A, dim))
969 if dim != 1
970 pdims = (dim, setdiff(1:ndims(A), dim)...) # put the selected dimension first
971 Ap = permutedims(A, pdims)
972 Av = vec(Ap)
973 sort_chunks!(Av, n, alg, order)
974 permutedims(Ap, invperm(pdims))
975 else
976 Av = A[:]
977 sort_chunks!(Av, n, alg, order)
978 reshape(Av, axes(A))
979 end
980 end
981
982 @noinline function sort_chunks!(Av, n, alg, order)
983 inds = LinearIndices(Av)
984 for s = first(inds):n:last(inds)
985 sort!(Av, s, s+n-1, alg, order)
986 end
987 Av
988 end

1001 const Floats = Union{Float32,Float64}
1002
1003 struct Left <: Ordering end
1004 struct Right <: Ordering end
1005
1006 left(::DirectOrdering) = Left()
1007 right(::DirectOrdering) = Right()
1008
1009 left(o::Perm) = Perm(left(o.order), o.data)
1010 right(o::Perm) = Perm(right(o.order), o.data)
1011
1012 lt(::Left, x::T, y::T) where {T<:Floats} = slt_int(y, x)
1013 lt(::Right, x::T, y::T) where {T<:Floats} = slt_int(x, y)
1014
1015 isnan(o::DirectOrdering, x::Floats) = (x!=x)
1016 isnan(o::Perm, i::Int) = isnan(o.order,o.data[i])
1017
1018 function nans2left!(v::AbstractVector, o::Ordering, lo::Int=first(axes(v,1)), hi::Int=last(axes(v,1)))
1019 i = lo
1020 @inbounds while i <= hi && isnan(o,v[i])
1021 i += 1
1022 end
1023 j = i + 1
1024 @inbounds while j <= hi
1025 if isnan(o,v[j])
1026 v[i], v[j] = v[j], v[i]
1027 i += 1
1028 end
1029 j += 1
1030 end
1031 return i, hi
1032 end
1033 function nans2right!(v::AbstractVector, o::Ordering, lo::Int=first(axes(v,1)), hi::Int=last(axes(v,1)))
1034 i = hi
1035 @inbounds while lo <= i && isnan(o,v[i])
1036 i -= 1
1037 end
1038 j = i - 1
1039 @inbounds while lo <= j
1040 if isnan(o,v[j])
1041 v[i], v[j] = v[j], v[i]
1042 i -= 1
1043 end
1044 j -= 1
1045 end
1046 return lo, i
1047 end
1048
1049 nans2end!(v::AbstractVector, o::ForwardOrdering) = nans2right!(v,o)
1050 nans2end!(v::AbstractVector, o::ReverseOrdering) = nans2left!(v,o)
1051 nans2end!(v::AbstractVector{Int}, o::Perm{<:ForwardOrdering}) = nans2right!(v,o)
1052 nans2end!(v::AbstractVector{Int}, o::Perm{<:ReverseOrdering}) = nans2left!(v,o)
1053
1054 issignleft(o::ForwardOrdering, x::Floats) = lt(o, x, zero(x))
1055 issignleft(o::ReverseOrdering, x::Floats) = lt(o, x, -zero(x))
1056 issignleft(o::Perm, i::Int) = issignleft(o.order, o.data[i])
1057
1058 function fpsort!(v::AbstractVector, a::Algorithm, o::Ordering)
1059 i, j = lo, hi = nans2end!(v,o)
1060 @inbounds while true
1061 while i <= j && issignleft(o,v[i]); i += 1; end
1062 while i <= j && !issignleft(o,v[j]); j -= 1; end
1063 i <= j || break
1064 v[i], v[j] = v[j], v[i]
1065 i += 1; j -= 1
1066 end
1067 sort!(v, lo, j, a, left(o))
1068 sort!(v, i, hi, a, right(o))
1069 return v
1070 end

1073 fpsort!(v::AbstractVector, a::Sort.PartialQuickSort, o::Ordering) =
1074 sort!(v, first(axes(v,1)), last(axes(v,1)), a, o)
1075
1076 sort!(v::AbstractVector{<:Floats}, a::Algorithm, o::DirectOrdering) = fpsort!(v,a,o)
1077 sort!(v::Vector{Int}, a::Algorithm, o::Perm{<:DirectOrdering,<:Vector{<:Floats}}) = fpsort!(v,a,o)
