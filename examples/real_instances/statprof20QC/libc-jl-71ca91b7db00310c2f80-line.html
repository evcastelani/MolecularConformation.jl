21 # Wrapper for an OS file descriptor (on both Unix and Windows)
22 """
23 RawFD
24
25 Primitive type which wraps the native OS file descriptor.
26 `RawFD`s can be passed to methods like [`stat`](@ref) to
27 discover information about the underlying file, and can
28 also be used to open streams, with the `RawFD` describing
29 the OS file backing the stream.
30 """
31 primitive type RawFD 32 end
32 RawFD(fd::Integer) = bitcast(RawFD, Cint(fd))
33 RawFD(fd::RawFD) = fd
34 Base.cconvert(::Type{Cint}, fd::RawFD) = bitcast(Cint, fd)
35
36 dup(x::RawFD) = ccall((@static Sys.iswindows() ? :_dup : :dup), RawFD, (RawFD,), x)
37 dup(src::RawFD, target::RawFD) = systemerror("dup", -1 ==
38 ccall((@static Sys.iswindows() ? :_dup2 : :dup2), Int32,
39 (RawFD, RawFD), src, target))

45 WindowsRawSocket(handle::WindowsRawSocket) = handle

48 _get_osfhandle(fd::RawFD) = ccall(:_get_osfhandle, WindowsRawSocket, (RawFD,), fd)
49 _get_osfhandle(fd::WindowsRawSocket) = fd

51 new_handle = Ref(WindowsRawSocket(Ptr{Cvoid}(-1)))

53 DUPLICATE_SAME_ACCESS = 0x2

56 my_process, src, my_process, new_handle, 0, false, DUPLICATE_SAME_ACCESS)

191 """
192 strptime([format], timestr)
193
194 Parse a formatted time string into a `TmStruct` giving the seconds, minute, hour, date, etc.
195 Supported formats are the same as those in the standard C library. On some platforms,
196 timezones will not be parsed correctly. If the result of this function will be passed to
197 `time` to convert it to seconds since the epoch, the `isdst` field should be filled in
198 manually. Setting it to `-1` will tell the C library to use the current system settings to
199 determine the timezone.
200 """
201 strptime(timestr::AbstractString) = strptime("%c", timestr)
202 function strptime(fmt::AbstractString, timestr::AbstractString)
203 tm = TmStruct()
204 r = ccall(:strptime, Cstring, (Cstring, Cstring, Ref{TmStruct}), timestr, fmt, tm)
205 # the following would tell mktime() that this is a local time, and that
206 # it should try to guess the timezone. not sure if/how this should be
207 # exposed in the API.
208 # tm.isdst = -1
209 if r == C_NULL
210 # TODO: better error message
211 throw(ArgumentError("invalid arguments"))
212 end
213 @static if Sys.isapple()
214 # if we didn't explicitly parse the weekday or year day, use mktime
215 # to fill them in automatically.
216 if !occursin(r"([^%]|^)%(a|A|j|w|Ow)", fmt)
217 ccall(:mktime, Int, (Ref{TmStruct},), tm)
218 end
219 end
220 return tm
221 end

306 function FormatMessage(e=GetLastError())
307 FORMAT_MESSAGE_ALLOCATE_BUFFER = UInt32(0x100)
308 FORMAT_MESSAGE_FROM_SYSTEM = UInt32(0x1000)
309 FORMAT_MESSAGE_IGNORE_INSERTS = UInt32(0x200)

311 lpMsgBuf = Ref{Ptr{UInt16}}()
312 lpMsgBuf[] = 0
313 len = ccall(:FormatMessageW, stdcall, UInt32, (Cint, Ptr{Cvoid}, Cint, Cint, Ptr{Ptr{UInt16}}, Cint, Ptr{Cvoid}),
314 FORMAT_MESSAGE_ALLOCATE_BUFFER | FORMAT_MESSAGE_FROM_SYSTEM | FORMAT_MESSAGE_IGNORE_INSERTS | FORMAT_MESSAGE_MAX_WIDTH_MASK,
315 C_NULL, e, 0, lpMsgBuf, 0, C_NULL)
316 p = lpMsgBuf[]
317 len == 0 && return ""
318 buf = Vector{UInt16}(undef, len)

320 ccall(:LocalFree, stdcall, Ptr{Cvoid}, (Ptr{Cvoid},), p)

366 # To limit dependency on rand functionality implemented in the Random module,
367 # Libc.rand is used in file.jl, and could be used in error.jl (but it breaks a test)
368 """
369 rand([T::Type])
370
371 Interface to the C `rand()` function. If `T` is provided, generate a value of type `T`
372 by composing two calls to `rand()`. `T` can be `UInt32` or `Float64`.
373 """
374 rand() = ccall(:rand, Cint, ())
375 # RAND_MAX at least 2^15-1 in theory, but we assume 2^16-1 (in practice, it's 2^31-1)
376 rand(::Type{UInt32}) = ((rand() % UInt32) << 16) (rand() % UInt32)
377 rand(::Type{Float64}) = rand(UInt32) / 2^32
