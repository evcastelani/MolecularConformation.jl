53 clamp(x, lo, hi)
54
55 Return `x` if `lo <= x <= hi`. If `x > hi`, return `hi`. If `x < lo`, return `lo`. Arguments
56 are promoted to a common type.
57
58 # Examples
59 ```jldoctest
60 julia> clamp.([pi, 1.0, big(10.)], 2., 9.)

63 2.0
64 9.0
65

68 6
69 6
70 10
71 ```
72 """
73 clamp(x::X, lo::L, hi::H) where {X,L,H} =
74 ifelse(x > hi, convert(promote_type(X,L,H), hi),
75 ifelse(x < lo,

77 convert(promote_type(X,L,H), x)))

93 @horner(x, p...)
94 Evaluate p[1] + x * (p[2] + x * (....)), i.e. a polynomial via Horner's rule
95 """
96 macro horner(x, p...)
97 ex = esc(p[end])
98 for i = length(p)-1:-1:1
99 ex = :(muladd(t, $ex, $(esc(p[i]))))
100 end
101 Expr(:block, :(t = $(esc(x))), ex)
102 end

104 # Evaluate p[1] + z*p[2] + z^2*p[3] + ... + z^(n-1)*p[n]. This uses
105 # Horner's method if z is real, but for complex z it uses a more
106 # efficient algorithm described in Knuth, TAOCP vol. 2, section 4.6.4,
107 # equation (3).

109 """
110 @evalpoly(z, c...)
111
112 Evaluate the polynomial ``\\sum_k c[k] z^{k-1}`` for the coefficients `c[1]`, `c[2]`, ...;
113 that is, the coefficients are given in ascending order by power of `z`. This macro expands
114 to efficient inline code that uses either Horner's method or, for complex `z`, a more
115 efficient Goertzel-like algorithm.
116
117 # Examples
118 ```jldoctest
119 julia> @evalpoly(3, 1, 0, 1)
120 10
121
122 julia> @evalpoly(2, 1, 0, 1)
123 5
124
125 julia> @evalpoly(2, 1, 1, 1)
126 7
127 ```
128 """
129 macro evalpoly(z, p...)
130 a = :($(esc(p[end])))
131 b = :($(esc(p[end-1])))
132 as = []
133 for i = length(p)-2:-1:1
134 ai = Symbol("a", i)
135 push!(as, :($ai = $a))
136 a = :(muladd(r, $ai, $b))
137 b = :($(esc(p[i])) - s * $ai) # see issue #15985 on fused mul-subtract
138 end
139 ai = :a0
140 push!(as, :($ai = $a))
141 C = Expr(:block,
142 :(x = real(tt)),
143 :(y = imag(tt)),
144 :(r = x + x),
145 :(s = muladd(x, x, y*y)),
146 as...,
147 :(muladd($ai, tt, $b)))
148 R = Expr(:macrocall, Symbol("@horner"), (), :tt, map(esc, p)...)
149 :(let tt = $(esc(z))
150 isa(tt, Complex) ? $C : $R
151 end)
152 end

540 function hypot(x::T, y::T) where T<:Number
541 ax = abs(x)
542 ay = abs(y)
543 if ax < ay
544 ax, ay = ay, ax
545 end
546 if iszero(ax)
547 r = ay / oneunit(ax)
548 else
549 r = ay / ax
550 end
551
552 rr = ax * sqrt(1 + r * r)
553
554 # Use type of rr to make sure that return type is the same for
555 # all branches
556 if isnan(r)
557 isinf(ax) && return oftype(rr, Inf)
558 isinf(ay) && return oftype(rr, Inf)
559 return oftype(rr, r)
560 else
561 return rr
562 end
563 end

587 """
588 ldexp(x, n)
589
590 Compute ``x \\times 2^n``.
591
592 # Examples
593 ```jldoctest
594 julia> ldexp(5., 2)
595 20.0
596 ```
597 """
598 function ldexp(x::T, e::Integer) where T<:IEEEFloat
599 xu = reinterpret(Unsigned, x)
600 xs = xu & ~sign_mask(T)
601 xs >= exponent_mask(T) && return x # NaN or Inf
602 k = Int(xs >> significand_bits(T))
603 if k == 0 # x is subnormal
604 xs == 0 && return x # +-0
605 m = leading_zeros(xs) - exponent_bits(T)
606 ys = xs << unsigned(m)
607 xu = ys | (xu & sign_mask(T))
608 k = 1 - m
609 # underflow, otherwise may have integer underflow in the following n + k
610 e < -50000 && return flipsign(T(0.0), x)
611 end
612 # For cases where e of an Integer larger than Int make sure we properly
613 # overflow/underflow; this is optimized away otherwise.
614 if e > typemax(Int)
615 return flipsign(T(Inf), x)
616 elseif e < typemin(Int)
617 return flipsign(T(0.0), x)
618 end
619 n = e % Int
620 k += n
621 # overflow, if k is larger than maximum possible exponent
622 if k >= exponent_raw_max(T)
623 return flipsign(T(Inf), x)
624 end
625 if k > 0 # normal case
626 xu = (xu & ~exponent_mask(T)) | (rem(k, uinttype(T)) << significand_bits(T))
627 return reinterpret(T, xu)
628 else # subnormal case
629 if k <= -significand_bits(T) # underflow
630 # overflow, for the case of integer overflow in n + k
631 e > 50000 && return flipsign(T(Inf), x)
632 return flipsign(T(0.0), x)
633 end
634 k += significand_bits(T)
635 z = T(2.0)^-significand_bits(T)
636 xu = (xu & ~exponent_mask(T)) | (rem(k, uinttype(T)) << significand_bits(T))
637 return z*reinterpret(T, xu)
638 end
639 end

642 """
643 exponent(x) -> Int
644
645 Get the exponent of a normalized floating-point number.
646 """
647 function exponent(x::T) where T<:IEEEFloat
648 @noinline throw1(x) = throw(DomainError(x, "Cannot be NaN or Inf."))
649 @noinline throw2(x) = throw(DomainError(x, "Cannot be subnormal converted to 0."))
650 xs = reinterpret(Unsigned, x) & ~sign_mask(T)
651 xs >= exponent_mask(T) && throw1(x)
652 k = Int(xs >> significand_bits(T))
653 if k == 0 # x is subnormal
654 xs == 0 && throw2(x)
655 m = leading_zeros(xs) - exponent_bits(T)
656 k = 1 - m
657 end
658 return k - exponent_bias(T)
659 end

661 """
662 significand(x)
663
664 Extract the `significand(s)` (a.k.a. mantissa), in binary representation, of a
665 floating-point number. If `x` is a non-zero finite number, then the result will be
666 a number of the same type on the interval ``[1,2)``. Otherwise `x` is returned.
667
668 # Examples
669 ```jldoctest
670 julia> significand(15.2)/15.2
671 0.125
672
673 julia> significand(15.2)*8
674 15.2
675 ```
676 """
677 function significand(x::T) where T<:IEEEFloat
678 xu = reinterpret(Unsigned, x)
679 xs = xu & ~sign_mask(T)
680 xs >= exponent_mask(T) && return x # NaN or Inf
681 if xs <= (~exponent_mask(T) & ~sign_mask(T)) # x is subnormal
682 xs == 0 && return x # +-0
683 m = unsigned(leading_zeros(xs) - exponent_bits(T))
684 xs <<= m
685 xu = xs | (xu & sign_mask(T))
686 end
687 xu = (xu & ~exponent_mask(T)) | exponent_one(T)
688 return reinterpret(T, xu)
689 end

694 Return `(x,exp)` such that `x` has a magnitude in the interval ``[1/2, 1)`` or 0,
695 and `val` is equal to ``x \\times 2^{exp}``.
696 """
697 function frexp(x::T) where T<:IEEEFloat
698 xu = reinterpret(Unsigned, x)
699 xs = xu & ~sign_mask(T)
700 xs >= exponent_mask(T) && return x, 0 # NaN or Inf
701 k = Int(xs >> significand_bits(T))
702 if k == 0 # x is subnormal
703 xs == 0 && return x, 0 # +-0
704 m = leading_zeros(xs) - exponent_bits(T)
705 xs <<= unsigned(m)
706 xu = xs | (xu & sign_mask(T))
707 k = 1 - m
708 end
709 k -= (exponent_bias(T) - 1)
710 xu = (xu & ~exponent_mask(T)) | exponent_half(T)
711 return reinterpret(T, xu), k
712 end

800 function add22condh(xh::Float64, xl::Float64, yh::Float64, yl::Float64)
801 # This algorithm, due to Dekker, computes the sum of two
802 # double-double numbers and returns the high double. References:
803 # [1] http://www.digizeitschriften.de/en/dms/img/?PID=GDZPPN001170007
804 # [2] https://doi.org/10.1007/BF01397083
805 r = xh+yh
806 s = (abs(xh) > abs(yh)) ? (xh-r+yh+yl+xl) : (yh-r+xh+xl+yl)
807 zh = r+s
808 return zh
809 end
810
811 # multiples of pi/2, as double-double (ie with "tail")
812 const pi1o2_h = 1.5707963267948966 # convert(Float64, pi * BigFloat(1/2))
813 const pi1o2_l = 6.123233995736766e-17 # convert(Float64, pi * BigFloat(1/2) - pi1o2_h)
814
815 const pi2o2_h = 3.141592653589793 # convert(Float64, pi * BigFloat(1))
816 const pi2o2_l = 1.2246467991473532e-16 # convert(Float64, pi * BigFloat(1) - pi2o2_h)
817
818 const pi3o2_h = 4.71238898038469 # convert(Float64, pi * BigFloat(3/2))
819 const pi3o2_l = 1.8369701987210297e-16 # convert(Float64, pi * BigFloat(3/2) - pi3o2_h)
820
821 const pi4o2_h = 6.283185307179586 # convert(Float64, pi * BigFloat(2))
822 const pi4o2_l = 2.4492935982947064e-16 # convert(Float64, pi * BigFloat(2) - pi4o2_h)
823
824 """
825 rem2pi(x, r::RoundingMode)
826
827 Compute the remainder of `x` after integer division by `2π`, with the quotient rounded
828 according to the rounding mode `r`. In other words, the quantity
829
830 x - 2π*round(x/(2π),r)
831
832 without any intermediate rounding. This internally uses a high precision approximation of
833 2π, and so will give a more accurate result than `rem(x,2π,r)`
834
835 - if `r == RoundNearest`, then the result is in the interval ``[-π, π]``. This will generally
836 be the most accurate result. See also [`RoundNearest`](@ref).
837
838 - if `r == RoundToZero`, then the result is in the interval ``[0, 2π]`` if `x` is positive,.
839 or ``[-2π, 0]`` otherwise. See also [`RoundToZero`](@ref).
840
841 - if `r == RoundDown`, then the result is in the interval ``[0, 2π]``.
842 See also [`RoundDown`](@ref).
843 - if `r == RoundUp`, then the result is in the interval ``[-2π, 0]``.
844 See also [`RoundUp`](@ref).
845
846 # Examples
847 ```jldoctest
848 julia> rem2pi(7pi/4, RoundNearest)
849 -0.7853981633974485
850
851 julia> rem2pi(7pi/4, RoundDown)
852 5.497787143782138
853 ```
854 """
855 function rem2pi end
856 function rem2pi(x::Float64, ::RoundingMode{:Nearest})
857 abs(x) < pi && return x
858
859 n,y = rem_pio2_kernel(x)
860
861 if iseven(n)
862 if n & 2 == 2 # n % 4 == 2: add/subtract pi
863 if y.hi <= 0
864 return add22condh(y.hi,y.lo,pi2o2_h,pi2o2_l)
865 else
866 return add22condh(y.hi,y.lo,-pi2o2_h,-pi2o2_l)
867 end
868 else # n % 4 == 0: add 0
869 return y.hi+y.lo
870 end
871 else
872 if n & 2 == 2 # n % 4 == 3: subtract pi/2
873 return add22condh(y.hi,y.lo,-pi1o2_h,-pi1o2_l)
874 else # n % 4 == 1: add pi/2
875 return add22condh(y.hi,y.lo,pi1o2_h,pi1o2_l)
876 end
877 end
878 end
879 function rem2pi(x::Float64, ::RoundingMode{:ToZero})
880 ax = abs(x)
881 ax <= 2*Float64(pi,RoundDown) && return x
882
883 n,y = rem_pio2_kernel(x)
884
885 if iseven(n)
886 if n & 2 == 2 # n % 4 == 2: add pi
887 z = add22condh(y.hi,y.lo,pi2o2_h,pi2o2_l)
888 else # n % 4 == 0: add 0 or 2pi
889 if y.hi > 0
890 z = y.hi+y.lo
891 else # negative: add 2pi
892 z = add22condh(y.hi,y.lo,pi4o2_h,pi4o2_l)
893 end
894 end
895 else
896 if n & 2 == 2 # n % 4 == 3: add 3pi/2
897 z = add22condh(y.hi,y.lo,pi3o2_h,pi3o2_l)
898 else # n % 4 == 1: add pi/2
899 z = add22condh(y.hi,y.lo,pi1o2_h,pi1o2_l)
900 end
901 end
902 copysign(z,x)
903 end
904 function rem2pi(x::Float64, ::RoundingMode{:Down})
905 if x < pi4o2_h
906 if x >= 0
907 return x
908 elseif x > -pi4o2_h
909 return add22condh(x,0.0,pi4o2_h,pi4o2_l)
910 end
911 end
912
913 n,y = rem_pio2_kernel(x)
914
915 if iseven(n)
916 if n & 2 == 2 # n % 4 == 2: add pi
917 return add22condh(y.hi,y.lo,pi2o2_h,pi2o2_l)
918 else # n % 4 == 0: add 0 or 2pi
919 if y.hi > 0
920 return y.hi+y.lo
921 else # negative: add 2pi
922 return add22condh(y.hi,y.lo,pi4o2_h,pi4o2_l)
923 end
924 end
925 else
926 if n & 2 == 2 # n % 4 == 3: add 3pi/2
927 return add22condh(y.hi,y.lo,pi3o2_h,pi3o2_l)
928 else # n % 4 == 1: add pi/2
929 return add22condh(y.hi,y.lo,pi1o2_h,pi1o2_l)
930 end
931 end
932 end
933 function rem2pi(x::Float64, ::RoundingMode{:Up})
934 if x > -pi4o2_h
935 if x <= 0
936 return x
937 elseif x < pi4o2_h
938 return add22condh(x,0.0,-pi4o2_h,-pi4o2_l)
939 end
940 end
941
942 n,y = rem_pio2_kernel(x)
943
944 if iseven(n)
945 if n & 2 == 2 # n % 4 == 2: sub pi
946 return add22condh(y.hi,y.lo,-pi2o2_h,-pi2o2_l)
947 else # n % 4 == 0: sub 0 or 2pi
948 if y.hi < 0
949 return y.hi+y.lo
950 else # positive: sub 2pi
951 return add22condh(y.hi,y.lo,-pi4o2_h,-pi4o2_l)
952 end
953 end
954 else
955 if n & 2 == 2 # n % 4 == 3: sub pi/2
956 return add22condh(y.hi,y.lo,-pi1o2_h,-pi1o2_l)
957 else # n % 4 == 1: sub 3pi/2
958 return add22condh(y.hi,y.lo,-pi3o2_h,-pi3o2_l)
959 end
960 end
961 end
