117 function show(io::IO, level::LogLevel)
118 if level == BelowMinLevel print(io, "BelowMinLevel")
119 elseif level == Debug print(io, "Debug")
120 elseif level == Info print(io, "Info")
121 elseif level == Warn print(io, "Warn")
122 elseif level == Error print(io, "Error")
123 elseif level == AboveMaxLevel print(io, "AboveMaxLevel")
124 else print(io, "LogLevel($(level.level))")

132 _logmsg_docs = """
133 @debug message [key=value | value ...]
134 @info message [key=value | value ...]
135 @warn message [key=value | value ...]
136 @error message [key=value | value ...]
137
138 @logmsg level message [key=value | value ...]
139
140 Create a log record with an informational `message`. For convenience, four
141 logging macros `@debug`, `@info`, `@warn` and `@error` are defined which log at
142 the standard severity levels `Debug`, `Info`, `Warn` and `Error`. `@logmsg`
143 allows `level` to be set programmatically to any `LogLevel` or custom log level
144 types.
145
146 `message` should be an expression which evaluates to a string which is a human
147 readable description of the log event. By convention, this string will be
148 formatted as markdown when presented.
149
150 The optional list of `key=value` pairs supports arbitrary user defined
151 metadata which will be passed through to the logging backend as part of the
152 log record. If only a `value` expression is supplied, a key representing the
153 expression will be generated using `Symbol`. For example, `x` becomes `x=x`,
154 and `foo(10)` becomes `Symbol("foo(10)")=foo(10)`. For splatting a list of
155 key value pairs, use the normal splatting syntax, `@info "blah" kws...`.
156
157 There are some keys which allow automatically generated log data to be
158 overridden:
159
160 * `_module=mod` can be used to specify a different originating module from
161 the source location of the message.
162 * `_group=symbol` can be used to override the message group (this is
163 normally derived from the base name of the source file).
164 * `_id=symbol` can be used to override the automatically generated unique
165 message identifier. This is useful if you need to very closely associate
166 messages generated on different source lines.
167 * `_file=string` and `_line=integer` can be used to override the apparent
168 source location of a log message.
169
170 There's also some key value pairs which have conventional meaning:
171
172 * `maxlog=integer` should be used as a hint to the backend that the message
173 should be displayed no more than `maxlog` times.
174 * `exception=ex` should be used to transport an exception with a log message,
175 often used with `@error`. An associated backtrace `bt` may be attached
176 using the tuple `exception=(ex,bt)`.
177
178 # Examples
179
180 ```
181 @debug "Verbose debugging information. Invisible by default"
182 @info "An informational message"
183 @warn "Something was odd. You should pay attention"
184 @error "A non fatal error occurred"
185
186 x = 10
187 @info "Some variables attached to the message" x a=42.0
188
189 @debug begin
190 sA = sum(A)
191 "sum(A) = \$sA is an expensive operation, evaluated only when `shouldlog` returns true"
192 end
193
194 for i=1:10000
195 @info "With the default backend, you will only see (i = \$i) ten times" maxlog=10
196 @debug "Algorithm1" i progress=i/10000
197 end
198 ```
199 """

224 _log_record_ids = Set{Symbol}()
225 # Generate a unique, stable, short, somewhat human readable identifier for a
226 # logging *statement*. The idea here is to have a key against which log events
227 # can be filtered and otherwise manipulated. The key should uniquely identify
228 # the source location in the originating module, but ideally should be stable
229 # across versions of the originating module, provided the log generating
230 # statement itself doesn't change.
231 function log_record_id(_module, level, message, log_kws)
232 modname = _module === nothing ? "" : join(fullname(_module), "_")
233 # Use an arbitriraly chosen eight hex digits here. TODO: Figure out how to
234 # make the id exactly the same on 32 and 64 bit systems.
235 h = UInt32(hash(string(modname, level, message, log_kws)) & 0xFFFFFFFF)
236 while true
237 id = Symbol(modname, '_', string(h, base = 16, pad = 8))
238 # _log_record_ids is a registry of log record ids for use during
239 # compilation, to ensure uniqueness of ids. Note that this state will
240 # only persist during module compilation so it will be empty when a
241 # precompiled module is loaded.
242 if !(id in _log_record_ids)
243 push!(_log_record_ids, id)
244 return id
245 end
246 h += 1
247 end
248 end
