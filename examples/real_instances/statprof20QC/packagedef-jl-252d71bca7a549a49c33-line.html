208 ##
209 ## The inputs are sets of expressions found in each file.
210 ## Some of those expressions will generate methods which are identified via their signatures.
211 ## From "old" expressions we know their corresponding signatures, but from "new"
212 ## expressions we have not yet computed them. This makes old and new asymmetric.
213 ##
214 ## Strategy:
215 ## - For every old expr not found in the new ones,
216 ## + delete the corresponding methods (using the signatures we've previously computed)
217 ## + remove the sig entries from CodeTracking.method_info (")
218 ## Best to do all the deletion first (across all files and modules) in case a method is
219 ## simply being moved from one file to another.
220 ## - For every new expr found among the old ones,
221 ## + update the location info in CodeTracking.method_info
222 ## - For every new expr not found in the old ones,
223 ## + eval the expr
224 ## + extract signatures
225 ## + add to the ModuleExprsSigs
226 ## + add to CodeTracking.method_info
227 ##
228 ## Interestingly, the ex=>sigs link may not be the same as the sigs=>ex link.
229 ## Consider a conditional block,
230 ## if Sys.islinux()
231 ## f() = 1
232 ## g() = 2
233 ## else
234 ## g() = 3
235 ## end
236 ## From the standpoint of Revise's diff-and-patch functionality, we should look for
237 ## diffs in this entire block. (Really good backedge support---or a variant of `lower` that
238 ## links back to the specific expression---might change this, but for
239 ## now this is the right strategy.) From the standpoint of CodeTracking, we should
240 ## link the signature to the actual method-defining expression (either :(f() = 1) or :(g() = 2)).
241

243 get_method_from_match(mm::Core.MethodMatch) = mm.method
244 else
245 get_method_from_match(mm::Core.SimpleVector) = mm[3]::Method
246 end

248 function delete_missing!(exs_sigs_old::ExprsSigs, exs_sigs_new)
249 with_logger(_debug_logger) do
250 for (ex, sigs) in exs_sigs_old
251 haskey(exs_sigs_new, ex) && continue
252 # ex was deleted
253 sigs === nothing && continue
254 for sig in sigs
255 ret = Base._methods_by_ftype(sig, -1, typemax(UInt))
256 success = false
257 if !isempty(ret)
258 m = get_method_from_match(ret[end]) # the last method returned is the least-specific that matches, and thus most likely to be type-equal
259 methsig = m.sig
260 if sig <: methsig && methsig <: sig
261 locdefs = get(CodeTracking.method_info, sig, nothing)
262 if isa(locdefs, Vector{Tuple{LineNumberNode,Expr}})
263 if length(locdefs) > 1
264 # Just delete this reference but keep the method
265 line = firstline(ex)
266 ld = map(pr->linediff(line, pr[1]), locdefs)
267 idx = argmin(ld)
268 @assert ld[idx] < typemax(eltype(ld))
269 deleteat!(locdefs, idx)
270 continue
271 else
272 @assert length(locdefs) == 1
273 end
274 end
275 @debug "DeleteMethod" _group="Action" time=time() deltainfo=(sig, MethodSummary(m))
276 # Delete the corresponding methods
277 for p in workers()
278 try # guard against serialization errors if the type isn't defined on the worker
279 remotecall(Core.eval, p, Main, :(delete_method_by_sig($sig)))
280 catch
281 end
282 end
283 Base.delete_method(m)
284 # Remove the entries from CodeTracking data
285 delete!(CodeTracking.method_info, sig)
286 # Remove frame from JuliaInterpreter, if applicable. Otherwise debuggers
287 # may erroneously work with outdated code (265-like problems)
288 if haskey(JuliaInterpreter.framedict, m)
289 delete!(JuliaInterpreter.framedict, m)
290 end
291 if isdefined(m, :generator)
292 # defensively delete all generated functions
293 empty!(JuliaInterpreter.genframedict)
294 end
295 success = true
296 end
297 end
298 if !success
299 @debug "FailedDeletion" _group="Action" time=time() deltainfo=(sig,)
300 end
301 end
302 end
303 end
304 return exs_sigs_old
305 end

394
395 Create a cache for storing information about method definitions.
396 Adding signatures to such an object inserts them into `CodeTracking.method_info`,
397 which maps signature Tuple-types to `(lnn::LineNumberNode, ex::Expr)` pairs.
398 Because method signatures are unique within a module, this is the foundation for
399 identifying methods in a manner independent of source-code location.
400
401 It also has the following fields:
402
403 - `exprstack`: used when descending into `@eval` statements (via `push_expr` and `pop_expr!`)
404 `ex` (used in creating the `CodeTrackingMethodInfo` object) is the first entry in the stack.
405 - `allsigs`: a list of all method signatures defined by a given expression
406 - `deps`: list of top-level named objects (`Symbol`s and `GlobalRef`s) that method definitions
407 in this block depend on. For example, `if Sys.iswindows() f() = 1 else f() = 2 end` would
408 store `Sys.iswindows` here.
409 - `includes`: a list of `module=>filename` for any `include` statements encountered while the
410 expression was parsed.
411 """

523 function init_watching(pkgdata::PkgData, files=srcfiles(pkgdata))
524 udirs = Set{String}()
525 for file in files
526 dir, basename = splitdir(file)
527 dirfull = joinpath(basedir(pkgdata), dir)

541 end

546 updatetime!(watched_files[dirfull])
547 if !watching_files[]

556 """
557 revise_dir_queued(dirname)
558
559 Wait for one or more of the files registered in `Revise.watched_files[dirname]` to be
560 modified, and then queue the corresponding files on [`Revise.revision_queue`](@ref).
561 This is generally called via a [`Revise.TaskThunk`](@ref).
562 """
563 @noinline function revise_dir_queued(dirname)
564 @assert isabspath(dirname)
565 if !isdir(dirname)
566 sleep(0.1) # in case git has done a delete/replace cycle
567 end
568 stillwatching = true
569 while stillwatching
570 if !isdir(dirname)
571 with_logger(SimpleLogger(stderr)) do
572 @warn "$dirname is not an existing directory, Revise is not watching"
573 end
574 break
575 end
576
577 latestfiles, stillwatching = watch_files_via_dir(dirname) # will block here until file(s) change
578 for (file, id) in latestfiles
579 key = joinpath(dirname, file)
580 if key in keys(user_callbacks_by_file)
581 union!(user_callbacks_queue, user_callbacks_by_file[key])
582 notify(revision_event)
583 end
584 if id != NOPACKAGE
585 pkgdata = pkgdatas[id]
586 if hasfile(pkgdata, key) # issue #228
587 push!(revision_queue, (pkgdata, relpath(key, pkgdata)))
588 notify(revision_event)
589 end
590 end
591 end
592 end
593 return
594 end

597 """
598 revise_file_queued(pkgdata::PkgData, filename)
599
600 Wait for modifications to `filename`, and then queue the corresponding files on [`Revise.revision_queue`](@ref).
601 This is generally called via a [`Revise.TaskThunk`](@ref).
602
603 This is used only on platforms (like BSD) which cannot use [`Revise.revise_dir_queued`](@ref).
604 """
605 function revise_file_queued(pkgdata::PkgData, file)
606 if !isabspath(file)
607 file = joinpath(basedir(pkgdata), file)
608 end
609 if !file_exists(file)
610 sleep(0.1) # in case git has done a delete/replace cycle
611 end
612
613 dirfull, basename = splitdir(file)
614 stillwatching = true
615 while stillwatching
616 if !file_exists(file) && !isdir(file)
617 let file=file
618 with_logger(SimpleLogger(stderr)) do
619 @warn "$file is not an existing file, Revise is not watching"
620 end
621 end
622 notify(revision_event)
623 break
624 end
625 try
626 wait_changed(file) # will block here until the file changes
627 catch e
628 # issue #459
629 (isa(e, InterruptException) && throwto_repl(e)) || throw(e)
630 end
631
632 if file in keys(user_callbacks_by_file)
633 union!(user_callbacks_queue, user_callbacks_by_file[file])
634 notify(revision_event)
635 end
636
637 # Check to see if we're still watching this file
638 stillwatching = haskey(watched_files, dirfull)
639 PkgId(pkgdata) != NOPACKAGE && push!(revision_queue, (pkgdata, relpath(file, pkgdata)))
640 end
641 return
642 end

733 """
734 revise(; throw=false)
735
736 `eval` any changes in the revision queue. See [`Revise.revision_queue`](@ref).
737 If `throw` is `true`, throw any errors that occur during revision or callback;
738 otherwise these are only logged.
739 """
740 function revise(; throw=false)
741 sleep(0.01) # in case the file system isn't quite done writing out the new files
742 have_queue_errors = !isempty(queue_errors)
743
744 # Do all the deletion first. This ensures that a method that moved from one file to another
745 # won't get redefined first and deleted second.
746 revision_errors = Tuple{PkgData,String}[]
747 queue = sort!(collect(revision_queue); lt=pkgfileless)
748 finished = eltype(revision_queue)[]
749 mexsnews = ModuleExprsSigs[]
750 interrupt = false
751 for (pkgdata, file) in queue
752 try
753 push!(mexsnews, handle_deletions(pkgdata, file)[1])
754 push!(finished, (pkgdata, file))
755 catch err
756 throw && Base.throw(err)
757 interrupt |= isa(err, InterruptException)
758 push!(revision_errors, (pkgdata, file))
759 queue_errors[(pkgdata, file)] = (err, catch_backtrace())
760 end
761 end
762 # Do the evaluation
763 for ((pkgdata, file), mexsnew) in zip(finished, mexsnews)
764 defaultmode = PkgId(pkgdata).name == "Main" ? :evalmeth : :eval
765 i = fileindex(pkgdata, file)
766 i === nothing && continue # file was deleted by `handle_deletions`
767 fi = fileinfo(pkgdata, i)
768 modsremaining = Set(keys(mexsnew))
769 changed, err = true, nothing
770 while changed
771 changed = false
772 for (mod, exsnew) in mexsnew
773 mod ∈ modsremaining || continue
774 try
775 mode = defaultmode
776 # Allow packages to override the supplied mode
777 if isdefined(mod, :__revise_mode__)
778 mode = getfield(mod, :__revise_mode__)::Symbol
779 end
780 mode ∈ (:sigs, :eval, :evalmeth, :evalassign) || error("unsupported mode ", mode)
781 exsold = get(fi.modexsigs, mod, empty_exs_sigs)
782 for rex in keys(exsnew)
783 sigs, includes = eval_rex(rex, exsold, mod; mode=mode)
784 if sigs !== nothing
785 exsnew[rex] = sigs
786 end
787 if includes !== nothing
788 maybe_add_includes_to_pkgdata!(pkgdata, file, includes; eval_now=true)
789 end
790 end
791 delete!(modsremaining, mod)
792 changed = true
793 catch _err
794 err = _err
795 end
796 end
797 end
798 if isempty(modsremaining)
799 pkgdata.fileinfos[i] = FileInfo(mexsnew, fi)
800 delete!(queue_errors, (pkgdata, file))
801 else
802 throw && Base.throw(err)
803 interrupt |= isa(err, InterruptException)
804 push!(revision_errors, (pkgdata, file))
805 queue_errors[(pkgdata, file)] = (err, catch_backtrace())
806 end
807 end
808 if interrupt
809 for pkgfile in finished
810 haskey(queue_errors, pkgfile) || delete!(revision_queue, pkgfile)
811 end
812 else
813 empty!(revision_queue)
814 end
815 errors(revision_errors)
816 if !isempty(queue_errors)
817 if !have_queue_errors # only print on the first time errors occur
818 io = IOBuffer()
819 println(io, "\n") # better here than in the triple-quoted literal, see https://github.com/JuliaLang/julia/issues/34105
820 for (pkgdata, file) in keys(queue_errors)
821 println(io, " ", joinpath(basedir(pkgdata), file))
822 end
823 str = String(take!(io))
824 @warn """The running code does not match the saved version for the following files:$str
825 If the error was due to evaluation order, it can sometimes be resolved by calling `Revise.retry()`.
826 Use Revise.errors() to report errors again. Only the first error in each file is shown.
827 Your prompt color may be yellow until the errors are resolved."""
828 maybe_set_prompt_color(:warn)
829 end
830 else
831 maybe_set_prompt_color(:ok)
832 end
833 tracking_Main_includes[] && queue_includes(Main)
834
835 process_user_callbacks!(throw=throw)
836
837 nothing
838 end

930 """
931 includet(filename)
932
933 Load `filename` and track future changes. `includet` is intended for quick "user scripts"; larger or more
934 established projects are encouraged to put the code in one or more packages loaded with `using`
935 or `import` instead of using `includet`. See https://timholy.github.io/Revise.jl/stable/cookbook/
936 for tips about setting up the package workflow.
937
938 By default, `includet` only tracks modifications to *methods*, not *data*. See the extended help for details.
939 Note that this differs from packages, which evaluate all changes by default.
940 This default behavior can be overridden; see [Configuring the revise mode](@ref).
941
942 # Extended help
943
944 ## Behavior and justification for the default revision mode (`:evalmeth`)
945
946 `includet` uses a default `__revise_mode__ = :evalmeth`. The consequence is that if you change
947
948 ```
949 a = [1]
950 f() = 1
951 ```
952 to
953 ```
954 a = [2]
955 f() = 2
956 ```
957 then Revise will update `f` but not `a`.
958
959 This is the default choice for `includet` because such files typically mix method definitions and data-handling.
960 Data often has many untracked dependencies; later in the same file you might `push!(a, 22)`, but Revise cannot
961 determine whether you wish it to re-run that line after redefining `a`.
962 Consequently, the safest default choice is to leave the user in charge of data.
963
964 ## Workflow tips
965
966 If you have a series of computations that you want to run when you redefine your methods, consider separating
967 your method definitions from your computations:
968
969 - method definitions go in a package, or a file that you `includet` *once*
970 - the computations go in a separate file, that you re-`include` (no "t" at the end) each time you want to rerun
971 your computations.
972
973 This can be automated using [`entr`](@ref).
974
975 ## Internals
976
977 `includet` is essentially shorthand for
978
979 Revise.track(Main, filename; mode=:eval, skip_include=false)
980
981 Do *not* use `includet` for packages, as those should be handled by `using` or `import`.
982 If `using` and `import` aren't working, you may have packages in a non-standard location;
983 try fixing it with something like `push!(LOAD_PATH, "/path/to/my/private/repos")`.
984 (If you're working with code in Base or one of Julia's standard libraries, use
985 `Revise.track(mod)` instead, where `mod` is the module.)
986
987 `includet` is deliberately non-recursive, so if `filename` loads any other files,
988 they will not be automatically tracked.
989 (See [`Revise.track`](@ref) to set it up manually.)
990 """
991 function includet(mod::Module, file::AbstractString)
992 prev = Base.source_path(nothing)
993 if prev === nothing
994 file = abspath(file)
995 else
996 file = normpath(joinpath(dirname(prev), file))
997 end
998 tls = task_local_storage()
999 tls[:SOURCE_PATH] = file
1000 try
1001 track(mod, file; mode=:includet, skip_include=true)
1002 if prev === nothing
1003 delete!(tls, :SOURCE_PATH)
1004 else
1005 tls[:SOURCE_PATH] = prev
1006 end
1007 catch err
1008 if prev === nothing
1009 delete!(tls, :SOURCE_PATH)
1010 else
1011 tls[:SOURCE_PATH] = prev
1012 end
1013 if isa(err, ReviseEvalException)
1014 printstyled(stderr, "ERROR: "; color=Base.error_color());
1015 showerror(stderr, err; blame_revise=false)
1016 println(stderr, "\nin expression starting at ", err.loc)
1017 else
1018 throw(err)
1019 end
1020 end
1021 return nothing
1022 end

1402 const REVISE_ID = Base.PkgId(Base.UUID("295af30f-e4ad-537b-8983-00126c2a3abe"), "Revise")
1403 function watch_package_callback(id::PkgId)
1404 # `Base.package_callbacks` fire immediately after module initialization, and
1405 # would fire on Revise itself. This is not necessary for most users, and has
1406 # the downside that the user doesn't get to the REPL prompt until
1407 # `watch_package` finishes compiling. To prevent this, Revise hides the
1408 # actual `watch_package` method behind an `invokelatest`. This delays
1409 # compilation of everything that `watch_package` requires, leading to faster
1410 # perceived startup times.
1411 if id != REVISE_ID
1412 Base.invokelatest(watch_package, id)
1413 end
1414 return
1415 end
